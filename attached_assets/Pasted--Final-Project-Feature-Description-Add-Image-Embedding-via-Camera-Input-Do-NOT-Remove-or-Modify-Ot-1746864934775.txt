import cohere
import requests
import base64

co = cohere.ClientV2("<YOUR_API_KEY>")

image_base64 = f"data:{content_type};base64,{stringified_buffer}"

response = co.embed(
    model="embed-v4.0",
    input_type="image",
    embedding_types=["float"],
    images=[image_base64],
)

print(response)
